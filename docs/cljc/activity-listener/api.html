(defn idle-time
  [activity-id]
  (if-let [last-activity (common-state/get-state :activity-listener activity-id :last-activity)]
          #?(:clj  (- (time/epoch-ms) last-activity)
             :cljs (- (time/elapsed)  last-activity))))

idle-time-elapsed?
(defn idle-time-elapsed?
  [activity-id elapsed-ms]
  (if-let [idle-time (idle-time activity-id)]
          (and (-> elapsed-ms (number?))
               (-> elapsed-ms (<= idle-time)))))

lock-activity!
(defn lock-activity!
  [activity-id unlock-timeout]
  #?(:clj  (let [epoch-ms (time/epoch-ms)] (if (number? unlock-timeout) (common-state/assoc-state! :activity-listener activity-id :locked-until (+ unlock-timeout epoch-ms))))
     :cljs (let [elapsed  (time/elapsed)]  (if (number? unlock-timeout) (common-state/assoc-state! :activity-listener activity-id :locked-until (+ unlock-timeout elapsed))))))
